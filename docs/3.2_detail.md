javascriptfunction calcularImpacto(variacaoSelic, variacaoCambio) {
  const impactoSelic = (dividaCDI + dividaTJLP) * variacaoSelic;
  const impactoCambio = dividaUSD * variacaoCambio * taxaCambioAtual;
